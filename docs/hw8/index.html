Demand and Capacity Grouped Bars

library(vegawidget)

spec <- list(
  `$schema` = "https://vega.github.io/schema/vega-lite/v5.json",
  title = "Demand and Capacity",
  data = list(
    url = "https://calvin-data304.netlify.app/data/swd-lets-practice-ex-2-3.json"
  ),
  transform = list(
    list(
      calculate = "toNumber(replace(datum.capacity, ',', ''))",
      as = "capacity_num"
    ),
    list(
      calculate = "toNumber(replace(datum.demand, ',', ''))",
      as = "demand_num"
    ),
    list(
      fold = c("capacity_num", "demand_num"),
      as = c("Type", "Quantity")
    )
  ),
  mark = "bar",
  encoding = list(
    x = list(field = "date", type = "nominal"),
    xOffset = list(field = "Type"),
    y = list(field = "Quantity", type = "quantitative"),
    color = list(
      field = "Type",
      type = "nominal",
      scale = list(
        domain = c("capacity_num", "demand_num"),
        range = c("#1f77b4", "#aec7e8")
      )
    )
  )
)

as_vegaspec(spec)

Capacity and Unmet Demand Stacked bars

library(vegawidget)

spec <- list(
  `$schema` = "https://vega.github.io/schema/vega-lite/v5.json",
  title = "Capacity and Unmet Demand",
  data = list(
    url = "https://calvin-data304.netlify.app/data/swd-lets-practice-ex-2-3.json"
  ),
  transform = list(
    list(
      calculate = "toNumber(replace(datum.capacity, ',', ''))",
      as = "capacity_num"
    ),
    list(
      calculate = "toNumber(replace(datum.demand, ',', ''))",
      as = "demand_num"
    ),
    list(
      calculate = "max(datum.demand_num - datum.capacity_num, 0)",
      as = "unmet_demand"
    ),
    list(
      fold = c("capacity_num", "unmet_demand"),
      as = c("Type", "Quantity")
    )
  ),
  mark = "bar",
  encoding = list(
    x = list(field = "date", type = "nominal"),
    y = list(
      field = "Quantity",
      type = "quantitative",
      stack = "zero"
    ),
    color = list(
      field = "Type",
      type = "nominal",
      scale = list(
        domain = c("capacity_num", "unmet_demand"),
        range = c("#1f77b4", "#aec7e8")
      )
    )
  )
)
